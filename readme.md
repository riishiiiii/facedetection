# step 1 
create python virtual env 
cd to folder directory then
run `python3 -m venv env`

# step 2
activate env
run `source env/bin/activate`  (for max and linux)
run `source env/Script/activate` (for windows git bash)


# step 3
install packaged
run `pip install -r req.txt`
 
# step 4 
run script by
`python3 main.py`

# step 5
visit
`localhost:8001/docs`
